void PrimeGenerator() {
    vector<ll> prime;
    bool isprime[MAX + 1];
    memset(isprime, true, sizeof(isprime));
    for (ll j = 2; j <= MAX; j++) {
        if (isprime[j]) { 
            prime.push_back(j); 
            for (ll i = j * j; i <= MAX; i += j) {
                isprime[i] = false;
            }
        }
    }
}

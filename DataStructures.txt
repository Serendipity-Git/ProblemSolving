int parent[MAX];
int member[MAX];
for (int i=0;i<MAX;i++){
    parent[i]=-1;
    member[i]=1;
}
int findRoot(int n) {
	while (parent[n] != -1) { n = parent[n]; }
	return n;
}
void makeUnion(int r1, int r2) {
	if (r1 != r2) {
		if (member[r1] < member[r2]) { swap(r1, r2); }
		parent[r2] = r1;
		member[r1] += member[r2];
	}
}
